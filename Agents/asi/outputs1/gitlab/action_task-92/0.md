def navigate_to_project_members(project_id: str, settings_id: str, members_id: str):
    """Navigate from project page to members management page.
    
    Args:
        project_id: The ID of the project link
        settings_id: The ID of the Settings link
        members_id: The ID of the Members link
    
    Returns:
        None
    
    Examples:
        navigate_to_project_members('901', '395', '407')
    """
    click(project_id)  # Click project link
    click(settings_id)  # Click Settings link
    click(members_id)  # Click Members link

def invite_user_to_project(search_box_id: str, username: str, user_result_id: str, add_button_id: str):
    """Search for and invite a user to the current project.
    
    Args:
        search_box_id: The ID of the member search box
        username: The username to search for
        user_result_id: The ID of the user search result
        add_button_id: The ID of the "Add to project" button
    
    Returns:
        None
    
    Examples:
        invite_user_to_project('490', 'Abishek', '607', '512')
    """
    fill(search_box_id, username)  # Search for the user
    click(user_result_id)  # Click the user from search results
    click(add_button_id)  # Click "Add to project" button
