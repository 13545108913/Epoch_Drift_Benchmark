## Reusable Functions
```python
def navigate_to_issues(issues_link_id: str):
    """Navigate to the issues dashboard.
    
    Args:
        issues_link_id: The ID of the Issues link
        
    Returns:
        None
        
    Examples:
        navigate_to_issues('161')
    """
    click(issues_link_id)  # Click Issues link to go to issues dashboard


def search_issues(search_box_id: str, search_term: str):
    """Search for issues with a specific term in the search box.
    
    Args:
        search_box_id: The ID of the search box
        search_term: The term to search for in issues
        
    Returns:
        None
        
    Examples:
        search_issues('144', 'theme editor')
    """
    click(search_box_id)  # Click on the search box
    fill(search_box_id, search_term)  # Enter the search term
    keyboard_press("Enter")  # Press Enter to search


def apply_all_issues_filter(filter_button_id: str):
    """Apply the "All" filter to show all issues.
    
    Args:
        filter_button_id: The ID of the "All" filter button
        
    Returns:
        None
        
    Examples:
        apply_all_issues_filter('250')
    """
    click(filter_button_id)  # Click the "All" filter button
```
