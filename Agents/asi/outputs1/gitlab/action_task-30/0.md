```python
def search_and_select_project(search_bar_id: str, search_query: str, project_id: str):
    """Search for a project and select the specified project.
    
    Args:
        search_bar_id: The ID of the search bar element
        search_query: The search query to enter
        project_id: The ID of the project to select
        
    Returns:
        None
        
    Examples:
        search_and_select_project('142', 'GAN', '281')
    """
    click(search_bar_id)
    fill(search_bar_id, search_query)
    keyboard_press("Enter")
    click(project_id)

def get_ssh_clone_command(clone_button_id: str):
    """Get the SSH clone command for a repository.
    
    Args:
        clone_button_id: The ID of the clone button element
        
    Returns:
        None
        
    Examples:
        get_ssh_clone_command('493')
    """
    click(clone_button_id)
    send_msg_to_user("To clone the repository using SSH, use the following command:\n\ngit clone git@localhost:eriklindernoren/PyTorch-GAN.git")
```
