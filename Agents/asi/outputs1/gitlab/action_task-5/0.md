def search_repository(search_box_id: str, repo_name: str):
    """Search for a GitHub repository using the search box.
    
    Args:
        search_box_id: The ID of the search box element
        repo_name: The name of the repository to search for
        
    Returns:
        None
        
    Examples:
        search_repository('142', 'keycloak/keycloak')
        search_repository('142', 'microsoft/vscode')
    """
    click(search_box_id)
    fill(search_box_id, repo_name)
    keyboard_press("Enter")

def navigate_to_issues(issues_link_id: str):
    """Navigate to the issues page of the current repository.
    
    Args:
        issues_link_id: The ID of the Issues link
        
    Returns:
        None
        
    Examples:
        navigate_to_issues('256')
        navigate_to_issues('issues-tab')
    """
    click(issues_link_id)

def filter_issues_by_label(search_box_id: str, repo_name: str, label: str):
    """Filter repository issues by a specific label.
    
    Args:
        search_box_id: The ID of the search box in issues page
        repo_name: The name of the repository
        label: The label to filter by
        
    Returns:
        None
        
    Examples:
        filter_issues_by_label('208', 'keycloak/keycloak', 'flaky-test')
        filter_issues_by_label('208', 'microsoft/vscode', 'bug')
    """
    fill(search_box_id, f"{repo_name} label:{label}")
    keyboard_press("Enter")
