## Reusable Functions
```python
def navigate_to_repository(repo_link_id: str):
    """Navigate to a specific repository page.
    
    Args:
        repo_link_id: The ID of the repository link to click
        
    Returns:
        None
        
    Examples:
        navigate_to_repository('747')
    """
    click(repo_link_id)  # Click on the repository link


def apply_license_template(add_license_id: str, apply_template_id: str, mit_license_id: str):
    """Apply MIT license template to a repository.
    
    Args:
        add_license_id: The ID of the "Add license" button
        apply_template_id: The ID of the "Apply a license template" button
        mit_license_id: The ID of the MIT License option
        
    Returns:
        None
        
    Examples:
        apply_license_template('575', '497', '534')
    """
    click(add_license_id)  # Click "Add license"
    click(apply_template_id)  # Click "Apply a license template"
    click(mit_license_id)  # Select MIT License from dropdown


def commit_license_changes(commit_button_id: str):
    """Commit the license changes to the repository.
    
    Args:
        commit_button_id: The ID of the "Commit changes" button
        
    Returns:
        None
        
    Examples:
        commit_license_changes('669')
    """
    click(commit_button_id)  # Click "Commit changes"
```
