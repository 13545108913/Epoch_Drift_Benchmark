def navigate_to_profile_settings(username_id: str, settings_id: str):
    """Navigate from homepage to profile settings page.
    
    Args:
        username_id: The ID of the username/profile link
        settings_id: The ID of the Settings link
        
    Returns:
        None
        
    Examples:
        navigate_to_profile_settings('186', '202')
    """
    click(username_id)  # Click username/profile link
    click(settings_id)  # Click Settings link

def update_website_url(website_field_id: str, url: str, save_button_id: str):
    """Update website URL in profile settings and save changes.
    
    Args:
        website_field_id: The ID of the website URL input field
        url: The URL to set as homepage
        save_button_id: The ID of the save/update button
        
    Returns:
        None
        
    Examples:
        update_website_url('481', 'https://example.com', '512')
    """
    fill(website_field_id, url)  # Fill website URL field
    click(save_button_id)  # Click save/update button
