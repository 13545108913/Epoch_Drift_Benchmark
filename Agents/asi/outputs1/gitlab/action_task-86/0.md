```python
def create_new_repository(new_project_id: str, name_field_id: str, create_button_id: str, project_name: str):
    """Create a new empty repository with the given name.
    
    Args:
        new_project_id: The ID of the "New project" button/link
        name_field_id: The ID of the project name input field
        create_button_id: The ID of the create project button
        project_name: The name for the new repository
        
    Returns:
        None
        
    Examples:
        create_new_repository('223', '259', '310', 'chatgpt_plugin')
        create_new_repository('223', '259', '310', 'my_new_project')
    """
    click(new_project_id)  # Click "New project" to start repository creation
    fill(name_field_id, project_name)  # Fill in the repository name
    click(create_button_id)  # Click create button to finalize
```
