## Reusable Functions
```python
def search_and_select_repo(search_id: str | int, repo_name: str, repo_link_id: str | int):
    """Search for a repository and select it from search results.
    
    Args:
        search_id: The ID of the search input field
        repo_name: The name of the repository to search for
        repo_link_id: The ID of the repository link in search results
        
    Returns:
        None
        
    Examples:
        search_and_select_repo('142', 'cloud-to-butt', '313')
        search_and_select_repo('142', 'my-project', '315')
    """
    fill(search_id, repo_name)  # Fill search box with repository name
    press(search_id, 'Enter')  # Press Enter to search
    click(repo_link_id)  # Click on the repository link in search results
```

```python
def navigate_to_new_issue(issues_tab_id: str | int, new_issue_id: str | int):
    """Navigate from repository page to new issue creation page.
    
    Args:
        issues_tab_id: The ID of the Issues tab/link
        new_issue_id: The ID of the New Issue button/link
        
    Returns:
        None
        
    Examples:
        navigate_to_new_issue('282', '477')
        navigate_to_new_issue('285', '480')
    """
    click(issues_tab_id)  # Click on Issues tab
    click(new_issue_id)  # Click on New Issue button
```

```python
def create_issue_with_details(title_id: str | int, title: str, assign_id: str | int, due_date_id: str | int, due_date: str, submit_id: str | int):
    """Fill issue details and submit the issue.
    
    Args:
        title_id: The ID of the title input field
        title: The issue title text
        assign_id: The ID of the assign to me button
        due_date_id: The ID of the due date input field
        due_date: The due date in YYYY-MM-DD format
        submit_id: The ID of the submit button
        
    Returns:
        None
        
    Examples:
        create_issue_with_details('479', 'Fix bug', '587', '688', '2024-12-31', '694')
        create_issue_with_details('480', 'New feature', '590', '690', '2025-06-30', '695')
    """
    fill(title_id, title)  # Fill issue title
    click(assign_id)  # Assign to current user
    fill(due_date_id, due_date)  # Set due date
    click(submit_id)  # Submit the issue
```
