```python
def set_gitlab_status(profile_id: str, set_status_id: str, status_textbox_id: str, confirm_button_id: str, status: str):
    """Set GitLab user status to the specified status.
    
    Args:
        profile_id: ID of the user profile link
        set_status_id: ID of the "Set status" button
        status_textbox_id: ID of the status text input field
        confirm_button_id: ID of the confirm/save button
        status: The status text to set
        
    Returns:
        None
        
    Examples:
        set_gitlab_status('186', '195', '795', '801', 'Busy')
        set_gitlab_status('186', '195', '795', '801', 'Working from home')
    """
    click(profile_id)  # Click user profile link
    click(set_status_id)  # Click "Set status" button
    fill(status_textbox_id, status)  # Fill status text
    click(confirm_button_id)  # Click confirm/save button
```
