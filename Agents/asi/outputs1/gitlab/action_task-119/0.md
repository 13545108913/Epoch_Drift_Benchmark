```python
def navigate_to_milestones(issues_id: str, milestones_id: str):
    """Navigate from homepage to milestones page.
    
    Args:
        issues_id: The ID of the "Issues" link
        milestones_id: The ID of the "Milestones" link
        
    Returns:
        None
        
    Examples:
        navigate_to_milestones('282', '304')
    """
    click(issues_id)  # Click Issues link
    click(milestones_id)  # Click Milestones link

def create_milestone(new_milestone_id: str, title_field_id: str, start_date_field_id: str, 
                    due_date_field_id: str, create_button_id: str, title: str, 
                    start_date: str, end_date: str):
    """Create a new milestone with title and dates.
    
    Args:
        new_milestone_id: The ID of the "New milestone" link
        title_field_id: The ID of the title input field
        start_date_field_id: The ID of the start date input field
        due_date_field_id: The ID of the due date input field
        create_button_id: The ID of the "Create milestone" button
        title: The milestone title
        start_date: The start date in YYYY-MM-DD format
        end_date: The end date in YYYY-MM-DD format
        
    Returns:
        None
        
    Examples:
        create_milestone('495', '482', '553', '560', '564', 'Project Launch', '2024-01-01', '2024-01-31')
    """
    click(new_milestone_id)  # Click New milestone link
    fill(title_field_id, title)  # Fill milestone title
    fill(start_date_field_id, start_date)  # Fill start date
    fill(due_date_field_id, end_date)  # Fill due date
    click(create_button_id)  # Click Create milestone button
```
