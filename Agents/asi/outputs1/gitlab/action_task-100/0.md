```python
def sort_repos_by_stars(explore_id: str, sort_by_stars_id: str):
    """Navigate to explore projects and sort by star count.
    
    Args:
        explore_id: The ID of the "Explore projects" link
        sort_by_stars_id: The ID of the "Most stars" sorting option
        
    Returns:
        None
        
    Examples:
        sort_repos_by_stars('237', '275')
    """
    click(explore_id)  # Click Explore projects
    click(sort_by_stars_id)  # Click Most stars to sort
```

```python
def star_repository(star_button_id: str):
    """Star a single repository by clicking its star button.
    
    Args:
        star_button_id: The ID of the star button for a repository
        
    Returns:
        None
        
    Examples:
        star_repository('313')
        star_repository('344')
    """
    click(star_button_id)  # Click the star button
```
