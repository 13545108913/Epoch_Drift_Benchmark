```python
def search_project(search_box_id: str | int, project_name: str):
    """Search for a GitLab project using the search functionality.
    
    Args:
        search_box_id: The ID of the search input box
        project_name: The name of the project to search for
        
    Returns:
        None
        
    Examples:
        search_project('142', 'thoughtbot/administrate')
        search_project('142', 'rails/rails')
    """
    click(search_box_id)
    fill(search_box_id, project_name)
    press(search_box_id, 'Enter')
```

```python
def navigate_to_projects_page(projects_link_id: str | int):
    """Navigate to the main projects page in GitLab.
    
    Args:
        projects_link_id: The ID of the "Projects" link
        
    Returns:
        None
        
    Examples:
        navigate_to_projects_page('253')
        navigate_to_projects_page('projects_link')
    """
    click(projects_link_id)
```
