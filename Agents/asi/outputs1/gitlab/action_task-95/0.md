def navigate_to_repo_members(repo_id: str, settings_id: str, members_id: str):
    """Navigate from repository page to members management section.
    
    Args:
        repo_id: The ID of the repository link
        settings_id: The ID of the Settings link
        members_id: The ID of the Members link
    
    Returns:
        None
    
    Examples:
        navigate_to_repo_members('901', '395', '407')
    """
    click(repo_id)  # Click repository link
    click(settings_id)  # Click Settings link
    click(members_id)  # Click Members link

def invite_user_to_repo(search_field_id: str, user_name: str, user_result_id: str, add_button_id: str):
    """Search for and invite a user to the repository.
    
    Args:
        search_field_id: The ID of the member search field
        user_name: The name of the user to invite
        user_result_id: The ID of the user result in dropdown
        add_button_id: The ID of the "Add to project" button
    
    Returns:
        None
    
    Examples:
        invite_user_to_repo('490', 'Jakub Klinkovský', '723', '512')
    """
    fill(search_field_id, user_name)  # Fill search field with user name
    click(user_result_id)  # Click user result from dropdown
    click(add_button_id)  # Click add button to invite user
