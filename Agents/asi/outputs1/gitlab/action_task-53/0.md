def navigate_to_project_merge_requests(project_link_id: str, merge_requests_link_id: str):
    """Navigate from the main page to the merge requests page of a project.
    
    Args:
        project_link_id: The ID of the project link to click
        merge_requests_link_id: The ID of the "Merge Requests" link to click
        
    Returns:
        None
        
    Examples:
        navigate_to_project_merge_requests('286', '304')
    """
    click(project_link_id)  # Click on the project link
    click(merge_requests_link_id)  # Click on the "Merge Requests" link
```

def post_comment_to_mr(comment_text: str):
    """Post a comment to the currently open merge request.
    
    Args:
        comment_text: The text content to post as a comment
        
    Returns:
        None
        
    Examples:
        post_comment_to_mr('lgtm')
        post_comment_to_mr('Looks good to me')
    """
    comment_box = page.get_by_label("Comment")
    comment_box.fill(comment_text)  # Fill the comment box with the text
    submit_button = page.get_by_role("button", name="Add comment")
    submit_button.click()  # Click the submit button to post the comment
```
