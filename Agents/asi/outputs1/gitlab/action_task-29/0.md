```python
def search_project(search_box_id: str | int, project_name: str):
    """Search for a GitLab project using the search functionality.
    
    Args:
        search_box_id: The ID of the search box element
        project_name: The name of the project to search for
        
    Returns:
        None
        
    Examples:
        search_project('142', 'metaseq')
        search_project('search-box', 'my-project')
    """
    click(search_box_id)  # Click on the search box
    fill(search_box_id, project_name)  # Enter the project name
    keyboard_press('Enter')  # Execute the search
```

```python
def report_project_not_found(project_name: str):
    """Send a message to the user indicating the project was not found.
    
    Args:
        project_name: The name of the project that was not found
        
    Returns:
        None
        
    Examples:
        report_project_not_found('metaseq')
        report_project_not_found('my-project')
    """
    message = f"The search for '{project_name}' returned no results. There are no projects, issues, or other items matching this name in the current GitLab instance. Please verify the project name or check if you have access to it."
    send_msg_to_user(message)
```
