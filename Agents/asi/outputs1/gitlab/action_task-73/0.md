```python
def set_gitlab_status(profile_id: str, settings_id: str, status_textbox_id: str, update_button_id: str, status_text: str):
    """Set GitLab user status to specified text.
    
    Args:
        profile_id: ID of user profile dropdown
        settings_id: ID of Settings link in user menu
        status_textbox_id: ID of status text input field
        update_button_id: ID of Update profile settings button
        status_text: The status text to set
    
    Returns:
        None
    
    Examples:
        set_gitlab_status('186', '202', '400', '512', 'Out of Office')
        set_gitlab_status('186', '202', '400', '512', 'Working Remotely')
    """
    click(profile_id)  # Click user profile dropdown
    click(settings_id)  # Click Settings link
    fill(status_textbox_id, status_text)  # Fill status textbox
    click(update_button_id)  # Click update button to save
```
