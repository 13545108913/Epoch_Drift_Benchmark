def navigate_to_member_invitation(project_id: str, settings_id: str, members_id: str, invite_tab_id: str):
    """Navigate from project page to member invitation interface.
    
    Args:
        project_id: The ID of the project/repository link
        settings_id: The ID of the Settings link
        members_id: The ID of the Members link
        invite_tab_id: The ID of the "Invite member" tab
        
    Returns:
        None
        
    Examples:
        navigate_to_member_invitation('901', '395', '407', '474')
    """
    click(project_id)  # Click project/repository link
    click(settings_id)  # Click Settings link
    click(members_id)  # Click Members link
    click(invite_tab_id)  # Click "Invite member" tab

def invite_member_to_project(search_field_id: str, member_name: str, member_result_id: str, add_button_id: str):
    """Search for and invite a member to the project.
    
    Args:
        search_field_id: The ID of the member search field
        member_name: The name of the member to invite
        member_result_id: The ID of the member search result
        add_button_id: The ID of the "Add to project" button

        None

    click(member_result_id)  # Select member from search results
