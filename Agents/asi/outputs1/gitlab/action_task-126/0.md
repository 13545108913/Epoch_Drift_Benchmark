def search_and_select_project(search_bar_id: str, project_name: str, project_link_id: str):
    """Search for a project and select it from search results.
    
    Args:
        search_bar_id: The ID of the search bar element
        project_name: The name of the project to search for
        project_link_id: The ID of the project link in search results
        
    Returns:
        None
        
    Examples:
        search_and_select_project('142', 'awesome-python', '284')
    """
    click(search_bar_id)
    fill(search_bar_id, project_name)
    keyboard_press("Enter")
    click(project_link_id)

def navigate_to_issues_and_create_new(issues_link_id: str, new_issue_link_id: str):
    """Navigate to issues section and click new issue button.
    
    Args:
        issues_link_id: The ID of the Issues link
        new_issue_link_id: The ID of the New issue link
        
    Returns:
        None
        
    Examples:
        navigate_to_issues_and_create_new('278', '429')
    """
    click(issues_link_id)
    click(new_issue_link_id)

def create_issue(title_field_id: str, description_field_id: str, submit_button_id: str, title: str, description: str):
    """Fill issue details and submit the issue.
    
    Args:
        title_field_id: The ID of the title input field
        description_field_id: The ID of the description input field
        submit_button_id: The ID of the submit button
        title: The issue title text
        description: The issue description text
        
    Returns:
        None
        
    Examples:
        create_issue('419', '460', '505', 'Bug Report', 'Found a bug in the system')
    """
    fill(title_field_id, title)
    fill(description_field_id, description)
    click(submit_button_id)
