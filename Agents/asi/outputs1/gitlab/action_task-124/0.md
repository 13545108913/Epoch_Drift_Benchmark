```python
def navigate_to_new_issue_page(project_link_id: str, issues_link_id: str):
    """Navigate from homepage to the new issue creation page.
    
    Args:
        project_link_id: The ID of the project/repository link
        issues_link_id: The ID of the "Issues" link
        
    Returns:
        None
        
    Examples:
        navigate_to_new_issue_page('405', '253')
    """
    click(project_link_id)  # Click project/repository link
    click(issues_link_id)  # Click Issues link
    click('438')  # Click "New issue" button
```
