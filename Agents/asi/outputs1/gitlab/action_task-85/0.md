```python
def navigate_to_profile_settings(username_id: str):
    """Navigate from the current page to the profile settings page.
    
    Args:
        username_id: The ID of the username/profile link
        
    Returns:
        None
        
    Examples:
        navigate_to_profile_settings('186')
    """
    click(username_id)  # Click username/profile link
    click('202')  # Click Settings link

def update_profile_url(url_field_id: str, new_url: str, save_button_id: str):
    """Update the website URL in profile settings and save changes.
    
    Args:
        url_field_id: The ID of the website URL input field
        new_url: The new URL to set
        save_button_id: The ID of the save/update button
        
    Returns:
        None
        
    Examples:
        update_profile_url('481', 'byteblaze.github.io', '512')
    """
    fill(url_field_id, new_url)  # Fill the URL field with new URL
    click(save_button_id)  # Click save/update button
```
