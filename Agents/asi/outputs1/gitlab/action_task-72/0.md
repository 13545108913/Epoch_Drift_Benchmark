```python
def set_gitlab_status(profile_id: str, edit_button_id: str, status_textbox_id: str, set_button_id: str, status_text: str):
    """Set GitLab user status by navigating through profile and status editing interface.
    
    Args:
        profile_id: The ID of the user profile link/button
        edit_button_id: The ID of the "Edit status" button
        status_textbox_id: The ID of the status text input field
        set_button_id: The ID of the "Set status" button
        status_text: The status message to set
    
    Returns:
        None
    
    Examples:
        set_gitlab_status('186', '198', '798', '804', 'Working from home')
        set_gitlab_status('186', '198', '798', '804', 'On vacation')
    """
    click(profile_id)  # Click user profile to access settings
    click(edit_button_id)  # Click edit status button
    fill(status_textbox_id, status_text)  # Fill status text
    click(set_button_id)  # Click set status button to save
```
