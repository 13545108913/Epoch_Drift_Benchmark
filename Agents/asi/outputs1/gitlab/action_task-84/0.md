def navigate_to_profile_settings(username_id: str, settings_id: str):
    """Navigate from homepage to profile settings page.
    
    Args:
        username_id: The ID of the username link in top navigation
        settings_id: The ID of the Settings link
        
    Returns:
        None
        
    Examples:
        navigate_to_profile_settings('186', '202')
    """
    click(username_id)  # Click username link to access profile
    click(settings_id)  # Click Settings link to access profile settings

def update_website_url(website_field_id: str, new_url: str, save_button_id: str):
    """Update the website URL in profile settings and save changes.
    
    Args:
        website_field_id: The ID of the website URL input field
        new_url: The new URL to set as homepage
        save_button_id: The ID of the save/update button
        
    Returns:
        None
        
    Examples:
        update_website_url('481', 'www.example.com', '512')
    """
    fill(website_field_id, new_url)  # Update website URL field
    click(save_button_id)  # Save the changes
