```python
def navigate_to_repo_commits(repo_link_id: str):
    """Navigate to a repository's commit history page.
    
    Args:
        repo_link_id: The ID of the repository link to click
        
    Returns:
        None
        
    Examples:
        navigate_to_repo_commits('286')
    """
    click(repo_link_id)  # Click repository link
    click('580')  # Click commits link to access commit history
```

```python
def switch_to_main_branch():
    """Switch the current branch view to main branch.
    
    Args:
        None
        
    Returns:
        None
        
    Examples:
        switch_to_main_branch()
    """
    click('473')  # Click branch selector button
    click('1462')  # Click main branch option
```

```python
def scroll_commit_history(scroll_amount1: int, scroll_amount2: int):
    """Scroll through commit history to load and view commits.
    
    Args:
        scroll_amount1: First scroll amount
        scroll_amount2: Second scroll amount
        
    Returns:
        None
        
    Examples:
        scroll_commit_history(1000, 1500)
    """
    scroll(0, scroll_amount1)  # Scroll to load more commits
    scroll(0, scroll_amount2)  # Scroll further to view commit history
```
