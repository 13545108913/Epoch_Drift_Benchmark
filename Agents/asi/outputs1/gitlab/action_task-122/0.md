def navigate_to_new_issue(project_link_id: str, issues_link_id: str, new_issue_button_id: str):
    """Navigate from homepage to new issue creation page.
    
    Args:
        project_link_id: The ID of the project/repo link
        issues_link_id: The ID of the Issues link
        new_issue_button_id: The ID of the New issue button
        
    Returns:
        None
        
    Examples:
        navigate_to_new_issue('632', '252', '453')
    """
    click(project_link_id)  # Click project/repo link
    click(issues_link_id)  # Click Issues link
    click(new_issue_button_id)  # Click New issue button

def create_issue_with_details(title_field_id: str, title: str, due_date_field_id: str, due_date: str, submit_button_id: str):
    """Fill issue title, set due date, and submit the issue.
    
    Args:
        title_field_id: The ID of the title input field
        title: The issue title text
        due_date_field_id: The ID of the due date input field
        due_date: The due date in YYYY-MM-DD format
        submit_button_id: The ID of the submit button
        
    Returns:
        None
        
    Examples:
        create_issue_with_details('440', 'Bug fix', '649', '2024-12-31', '655')
    """
    fill(title_field_id, title)  # Fill issue title
    fill(due_date_field_id, due_date)  # Set due date
    click(submit_button_id)  # Submit the issue
