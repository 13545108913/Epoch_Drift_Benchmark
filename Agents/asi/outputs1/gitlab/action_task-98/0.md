def sort_projects_by_stars(explore_id: str, sort_by_stars_id: str):
    """Navigate to explore projects and sort by most stars.
    
    Args:
        explore_id: The ID of the "Explore projects" link
        sort_by_stars_id: The ID of the "Most stars" sort option
        
    Returns:
        None
        
    Examples:
        sort_projects_by_stars('237', '275')
    """
    click(explore_id)  # Click Explore projects link
    click(sort_by_stars_id)  # Click Most stars sort option

def star_repository(star_button_id: str):
    """Star a single repository.
    
    Args:
        star_button_id: The ID of the star button for a repository
        
    Returns:
        None
        
    Examples:
        star_repository('319')
        star_repository('350')
    """
    click(star_button_id)  # Click the star button
