def navigate_to_member_settings(project_id: str, settings_id: str, members_id: str):
    """Navigate from project page to member management settings.
    
    Args:
        project_id: The ID of the project link
        settings_id: The ID of the Settings link
        members_id: The ID of the Members link
        
    Returns:
        None
        
    Examples:
        navigate_to_member_settings('867', '395', '407')
    """
    click(project_id)  # Click on the project
    click(settings_id)  # Click on Settings
    click(members_id)  # Click on Members

def add_user_to_project(search_box_id: str, username: str, add_button_id: str):
    """Search for and add a user to the project.
    
    Args:
        search_box_id: The ID of the member search box
        username: The username to add
        add_button_id: The ID of the "Add to project" button
        
    Returns:
        None
        
    Examples:
        add_user_to_project('490', 'yjlou', '512')
    """
    fill(search_box_id, username)  # Search for the user
    keyboard_press("Enter")  # Select the user from search results
    click(add_button_id)  # Click "Add to project" button
