```python
def set_gitlab_status(profile_id: str, edit_button_id: str, status_textbox_id: str, set_button_id: str, status: str):
    """Set GitLab user status by navigating through profile menu.
    
    Args:
        profile_id: ID of the user profile link/button
        edit_button_id: ID of the "Edit status" button
        status_textbox_id: ID of the status text input field
        set_button_id: ID of the "Set status" button
        status: The status message to set
        
    Returns:
        None
        
    Examples:
        set_gitlab_status('186', '198', '798', '804', 'Enjoying life')
        set_gitlab_status('186', '198', '798', '804', 'Working from home')
    """
    click(profile_id)  # Click user profile to access menu
    click(edit_button_id)  # Click "Edit status" button
    fill(status_textbox_id, status)  # Fill status text with provided message
    click(set_button_id)  # Click "Set status" to save
```
