```python
def search_project(search_box_id: str | int, project_name: str):
    """Search for a project using the search box.
    
    Args:
        search_box_id: The ID of the search box element
        project_name: The name of the project to search for
        
    Returns:
        None
        
    Examples:
        search_project('142', 'primer design guide')
        search_project('123', 'web application')
    """
    fill(search_box_id, project_name)
    press(search_box_id, 'Enter')

def navigate_to_issues(issues_link_id: str | int):
    """Navigate to the issues page of a repository.
    
    Args:
        issues_link_id: The ID of the issues link element
        
    Returns:
        None
        
    Examples:
        navigate_to_issues('302')
        navigate_to_issues('450')
    """
    click(issues_link_id)

def assign_issue_to_self(assign_self_button_id: str | int, issue_title: str):
    """Assign an issue to yourself and send confirmation message.
    
    Args:
        assign_self_button_id: The ID of the assign yourself button
        issue_title: The title of the issue being assigned
        
    Returns:
        None
        
    Examples:
        assign_issue_to_self('769', 'Clarify usage of flash alert')
        assign_issue_to_self('880', 'Fix login bug')
    """
    click(assign_self_button_id)
    send_msg_to_user(f"The issue '{issue_title}' has been successfully assigned to you.")
```
