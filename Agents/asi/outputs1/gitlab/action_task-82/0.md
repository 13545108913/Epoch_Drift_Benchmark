```python
def update_gitlab_profile_url(profile_id: str, settings_id: str, url_field_id: str, update_button_id: str, url: str):
    """Update the homepage URL in GitLab profile settings.
    
    Args:
        profile_id: The ID of the profile dropdown/link
        settings_id: The ID of the Settings link
        url_field_id: The ID of the website URL input field
        update_button_id: The ID of the update/save button
        url: The homepage URL to set
        
    Returns:
        None
        
    Examples:
        update_gitlab_profile_url('186', '202', '481', '512', 'https://example.com')
    """
    click(profile_id)  # Click profile dropdown
    click(settings_id)  # Click Settings link
    fill(url_field_id, url)  # Fill website URL field
    click(update_button_id)  # Click update button
```
