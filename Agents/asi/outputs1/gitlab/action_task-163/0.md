```python
def search_and_select_repo(search_box_id: str, repo_name: str, repo_link_id: str):
    """Search for and select a repository from search results.
    
    Args:
        search_box_id: ID of the search box element
        repo_name: Name of the repository to search for
        repo_link_id: ID of the repository link in search results
        
    Returns:
        None
        
    Examples:
        search_and_select_repo('142', 'a11yproject', '278')
    """
    fill(search_box_id, repo_name)
    press(search_box_id, 'Enter')
    click(repo_link_id)

def navigate_to_new_issue(issues_link_id: str, new_issue_link_id: str):
    """Navigate from repository page to new issue creation form.
    
    Args:
        issues_link_id: ID of the Issues navigation link
        new_issue_link_id: ID of the New Issue button/link
        
    Returns:
        None
        
    Examples:
        navigate_to_new_issue('282', '491')
    """
    click(issues_link_id)
    click(new_issue_link_id)

def create_issue_with_details(title_field_id: str, title: str, assign_self_id: str, due_date_field_id: str, due_date: str, submit_button_id: str):
    """Fill issue details and submit a new issue.
    
    Args:
        title_field_id: ID of the issue title field
        title: Issue title text
        assign_self_id: ID of the "Assign to me" button
        due_date_field_id: ID of the due date field
        due_date: Due date in string format
        submit_button_id: ID of the submit button
        
    Returns:
        None
        
    Examples:
        create_issue_with_details('478', '404 for many URLs', '586', '687', '2030-1-3', '693')
    """
    fill(title_field_id, title)
    click(assign_self_id)
    fill(due_date_field_id, due_date)
    click(submit_button_id)
```
