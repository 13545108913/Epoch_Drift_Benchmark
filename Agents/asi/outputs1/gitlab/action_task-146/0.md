def create_from_template(new_project_id: str, template_tab_id: str, template_name: str, template_id: str, use_template_id: str):
    """Navigate to template selection and choose a specific template.
    
    Args:
        new_project_id: ID of the "New project" button/link
        template_tab_id: ID of the "Create from template" tab
        template_name: Name of the template to search for/select
        template_id: ID of the specific template element
        use_template_id: ID of the "Use template" button/radio
    
    Returns:
        None
    
    Examples:
        create_from_template('223', '242', 'NodeJS Express', '357', '358')
    """
    click(new_project_id)  # Click New project
    click(template_tab_id)  # Click Create from template tab
    click(template_id)  # Click specific template
    click(use_template_id)  # Click Use template

def create_project_with_name(project_name_field_id: str, project_name: str, create_button_id: str):
    """Fill project name and create the project.
    
    Args:
        project_name_field_id: ID of the project name input field
        project_name: Name for the new project
        create_button_id: ID of the Create project button
    
    Returns:
        None
    
    Examples:
        create_project_with_name('543', 'web_agent_nodejs', '586')
    """
    fill(project_name_field_id, project_name)  # Fill project name
    click(create_button_id)  # Click Create project
