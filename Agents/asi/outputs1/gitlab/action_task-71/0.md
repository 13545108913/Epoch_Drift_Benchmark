```python
def set_gitlab_status(profile_id: str, edit_button_id: str, status_textbox_id: str, set_button_id: str, status: str):
    """Set GitLab status message.
    
    Args:
        profile_id: ID of user profile link
        edit_button_id: ID of edit status button
        status_textbox_id: ID of status text input field
        set_button_id: ID of set status button
        status: Status message to set
        
    Returns:
        None
        
    Examples:
        set_gitlab_status('186', '198', '798', '804', 'Working from home')
        set_gitlab_status('186', '198', '798', '804', 'In a meeting')
    """
    click(profile_id)  # Click user profile link
    click(edit_button_id)  # Click edit status button
    fill(status_textbox_id, status)  # Enter status message
    click(set_button_id)  # Click set status button
```
