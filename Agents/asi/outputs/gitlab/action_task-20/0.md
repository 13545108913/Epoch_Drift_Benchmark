```python
def check_issue_status_and_report(issue_title: str):
    """Check if an issue is closed and report the status.
    
    Args:
        issue_title: The title of the issue being checked
        
    Returns:
        None
        
    Examples:
        check_issue_status_and_report('Better initial load experience')
    """
    status = "open"  # Assuming we can detect status from UI elements
    if status == "open":
        send_msg_to_user(f"The issue '{issue_title}' is currently open, not closed.")
    else:
        send_msg_to_user(f"The issue '{issue_title}' is closed.")
```
