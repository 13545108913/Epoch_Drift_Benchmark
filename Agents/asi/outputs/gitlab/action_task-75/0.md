```python
def edit_html_file(web_ide_id: str, file_id: str, file_content: str):
    """Open web IDE and edit an HTML file with new content.
    
    Args:
        web_ide_id: The ID of the Web IDE link/button
        file_id: The ID of the HTML file to edit
        file_content: The complete HTML content to replace the file with
        
    Returns:
        None
        
    Examples:
        edit_html_file('774', '751', '<html>...</html>')
    """
    click(web_ide_id)  # Click Web IDE link
    click(file_id)  # Click the HTML file
    keyboard_press("ControlOrMeta+a")  # Select all content
    fill("1404", file_content)  # Replace with new HTML content

def commit_changes(commit_button_id: str, confirm_button_id: str):
    """Commit changes to save and deploy updates.
    
    Args:
        commit_button_id: The ID of the Commit button
        confirm_button_id: The ID of the confirmation button
        
    Returns:
        None
        
    Examples:
        commit_changes('791', '1794')
    """
    click(commit_button_id)  # Click Commit button
    click(confirm_button_id)  # Click confirmation button
```
