```python
def check_collaborator_status(collaborator_names: list, repo_name: str):
    """Check and report the status of collaborators on a repository.
    
    Args:
        collaborator_names: List of collaborator names to check
        repo_name: Name of the repository
        
    Returns:
        None
        
    Examples:
        check_collaborator_status(['Jakub Klinkovský', 'Benoît Blanchon'], 'gimmiethat.space')
    """
    all_collaborators = True
    for name in collaborator_names:
        # Check if collaborator exists in member list
        if not is_collaborator(name):
            all_collaborators = False
            break
    
    if all_collaborators:
        send_msg_to_user(f"All specified users are already collaborators on the {repo_name} repository with Developer roles. No action needed.")
    else:
        send_msg_to_user(f"Some users are not collaborators on {repo_name}. You can invite them from the member settings.")
```
