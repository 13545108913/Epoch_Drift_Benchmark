def filter_issues_by_label(search_box_id: str | int, label: str):
    """Filter issues by label using the search box.
    
    Args:
        search_box_id: The ID of the search box element
        label: The label to filter by (e.g., "bug", "enhancement")
    
    Returns:
        None
    
    Examples:
        filter_issues_by_label('655', 'bug')
        filter_issues_by_label('655', 'enhancement')
    """
    fill(search_box_id, f"label:{label}")
    keyboard_press("Enter")
