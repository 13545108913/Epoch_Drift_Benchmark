```python
def navigate_to_issues(issues_link_id: str):
    """Navigate to the issues page of the project.
    
    Args:
        issues_link_id: The ID of the issues navigation link
        
    Returns:
        None
        
    Examples:
        navigate_to_issues('350')
        navigate_to_issues('issues_tab')
    """
    click(issues_link_id)  # Click on issues navigation link
```

```python
def filter_issues_by_label(search_input_id: str, label_filter: str):
    """Filter issues by a specific label using the search input.
    
    Args:
        search_input_id: The ID of the issues search input field
        label_filter: The label filter string to apply
        
    Returns:
        None
        
    Examples:
        filter_issues_by_label('611', 'label:"type: bug"')
        filter_issues_by_label('search_box', 'label:"enhancement"')
    """
    fill(search_input_id, label_filter)  # Enter label filter in search
    press(search_input_id, 'Enter')  # Apply the filter
```
