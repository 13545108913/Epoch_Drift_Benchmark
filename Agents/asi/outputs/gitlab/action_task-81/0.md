```python
def navigate_to_profile_settings(profile_link_id: str):
    """Navigate from preferences page to profile settings page.
    
    Args:
        profile_link_id: The ID of the "Profile" link
        
    Returns:
        None
        
    Examples:
        navigate_to_profile_settings('275')
    """

```

```python
def set_website_url(website_field_id: str, url: str, save_button_id: str):
    """Set website URL in profile settings and save changes.
    
    Args:
        website_field_id: The ID of the website URL input field
        url: The URL to set as homepage
        save_button_id: The ID of the save/update button
        
    Returns:
        None
        
    Examples:
        set_website_url('878', 'https://example.com', '914')
    """
    fill(website_field_id, url)  # Fill website URL field
    click(save_button_id)  # Click save/update button
```
