```python
def navigate_to_repo_members(repo_id: str, settings_id: str, members_id: str):
    """Navigate from repository page to members management page.
    
    Args:
        repo_id: The ID of the repository link
        settings_id: The ID of the Settings link
        members_id: The ID of the Members link
        
    Returns:
        None
        
    Examples:
        navigate_to_repo_members('483', '509', '493')
    """
    click(repo_id)  # Click repository link
    click(settings_id)  # Click Settings link
    click(members_id)  # Click Members link
```

```python
def invite_user_to_repo(username: str, role: str = "guest"):
    """Invite a user to the repository with specified role.
    
    Args:
        username: The username to invite
        role: The role to assign (default: "guest")
        
    Returns:
        None
        
    Examples:
        invite_user_to_repo('vinta', 'guest')
    """
    fill("member_search", username)  # Fill username in member search field
    click("invite_button")  # Click invite button
    select_option("role_select", role)  # Select role from dropdown
    click("confirm_invite")  # Confirm the invitation
```
