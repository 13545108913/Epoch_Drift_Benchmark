```python
def navigate_to_issues(project_link_id: str, issues_link_id: str):
    """Navigate from homepage to project issues page.
    
    Args:
        project_link_id: The ID of the project link
        issues_link_id: The ID of the issues link
        
    Returns:
        None
        
    Examples:
        navigate_to_issues('624', '347')
    """
    click(project_link_id)  # Click project link
    click(issues_link_id)  # Click issues link

def assign_issue_to_user(issue_link_id: str, edit_assignee_id: str, user_option_id: str):
    """Assign an issue to a specific user.
    
    Args:
        issue_link_id: The ID of the issue link
        edit_assignee_id: The ID of the edit assignee link
        user_option_id: The ID of the user option in dropdown
        
    Returns:
        None
        
    Examples:
        assign_issue_to_user('1118', '1198', '1449')
    """
    click(issue_link_id)  # Click on the issue
    click(edit_assignee_id)  # Click edit assignee
    click(user_option_id)  # Select user from dropdown
```
