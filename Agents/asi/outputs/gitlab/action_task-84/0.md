## Reusable Functions
```python
def navigate_to_profile_settings(profile_link_id: str):
    """Navigate to the profile settings page from the main page.
    
    Args:
        profile_link_id: The ID of the profile/preferences link
        
    Returns:
        None
        
    Examples:
        navigate_to_profile_settings('272')
    """
    click(profile_link_id)  # Click profile/preferences link
    click('275')  # Click Profile section
```

```python
def set_custom_homepage_url(website_field_id: str, update_button_id: str, url: str):
    """Set a custom URL in the website field and save the profile settings.
    
    Args:
        website_field_id: The ID of the website URL input field
        update_button_id: The ID of the update/save button
        url: The URL to set as homepage
        
    Returns:
        None
        
    Examples:
        set_custom_homepage_url('878', '914', 'www.example.com')
    """
    fill(website_field_id, url)  # Fill website URL field
    click(update_button_id)  # Click update button to save changes
```

```python
def close_dropdown_and_navigate(dropdown_id: str, navigation_target_id: str):
    """Close an open dropdown menu and navigate to a target section.
    
    Args:
        dropdown_id: The ID to click to close dropdown (or escape)
        navigation_target_id: The ID of the target section to navigate to
        
    Returns:
        None
        
    Examples:
        close_dropdown_and_navigate('269', '275')
    """
    keyboard_press("Escape")  # Press Escape to close dropdown
    click(navigation_target_id)  # Click target navigation element
```
