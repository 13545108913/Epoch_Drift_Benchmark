```python
def create_issue(repo_id: str, title: str, assign_to_me_id: str, due_date: str):
    """Create a new issue in a repository with title, self-assignment, and due date.
    
    Args:
        repo_id: The ID of the repository link
        title: The title of the issue
        assign_to_me_id: The ID of the "Assign to me" link
        due_date: The due date in YYYY-MM-DD format
        
    Returns:
        None
        
    Examples:
        create_issue('857', 'Bug fix needed', '734', '2024-12-31')
    """
    click(repo_id)  # Click repository link
    click('347')  # Click Issues link
    click('624')  # Click New issue link
    fill('592', title)  # Fill issue title
    click(assign_to_me_id)  # Assign to myself
    fill('838', due_date)  # Set due date
    click('841')  # Create issue
```

```python
def notify_issue_creation(issue_title: str, assignee: str, due_date: str):
    """Send success message after creating an issue.
    
    Args:
        issue_title: The title of the created issue
        assignee: The name of the assignee
        due_date: The due date that was set
        
    Returns:
        None
        
    Examples:
        notify_issue_creation('Documentation update', 'Alice', '2024-06-30')
    """
    message = f"Successfully created issue '{issue_title}'. The issue has been assigned to {assignee} with a due date of {due_date}."
    send_msg_to_user(message)
```
