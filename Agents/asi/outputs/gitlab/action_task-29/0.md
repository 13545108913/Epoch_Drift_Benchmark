def search_repository(search_bar_id: str | int, repo_name: str):
    """Search for a repository using the search bar.
    
    Args:
        search_bar_id: The ID of the search bar element
        repo_name: The name of the repository to search for
        
    Returns:
        None
        
    Examples:
        search_repository('156', 'metaseq')
        search_repository('303', 'react')
    """
    click(search_bar_id)
    fill(search_bar_id, repo_name)
    press(search_bar_id, 'Enter')

def navigate_to_clone_options(project_link_id: str | int):
    """Navigate to a project and open clone options.
    
    Args:

    Returns:
        None
        
    Examples:
        navigate_to_clone_options('341')
        navigate_to_clone_options('215')
    """
