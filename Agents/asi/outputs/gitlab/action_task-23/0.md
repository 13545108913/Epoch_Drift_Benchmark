```python
def search_commits(search_box_id: str, author: str, date: str):
    """Search for commits by author and date.
    
    Args:
        search_box_id: The ID of the search box element
        author: The author name to filter by
        date: The date to filter by (YYYY-MM-DD format)
    
    Returns:
        None
    
    Examples:
        search_commits('660', 'Eric', '2024-03-02')
        search_commits('660', 'John', '2024-01-15')
    """
    click(search_box_id)
    fill(search_box_id, f"author:{author} date:{date}")
```
