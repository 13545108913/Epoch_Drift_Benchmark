```python
def navigate_to_repo_contributors(repo_id: str, analytics_id: str, repo_analytics_id: str, contributors_id: str):
    """Navigate from repository page to contributors analytics page.
    
    Args:
        repo_id: ID of the repository link
        analytics_id: ID of the Analytics section link
        repo_analytics_id: ID of the Repository analytics link
        contributors_id: ID of the Contributors link
        
    Returns:
        None
        
    Examples:
        navigate_to_repo_contributors('513', '478', '479', '482')
    """
    click(repo_id)  # Click on repository
    click(analytics_id)  # Click Analytics section
    click(repo_analytics_id)  # Click Repository analytics
    click(contributors_id)  # Click Contributors page
```
