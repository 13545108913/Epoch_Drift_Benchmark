```python
def navigate_to_repository_analytics(project_link_id: str, analytics_link_id: str, repository_link_id: str):
    """Navigate from project page to repository analytics page.
    
    Args:
        project_link_id: The ID of the project link
        analytics_link_id: The ID of the Analytics link
        repository_link_id: The ID of the Repository link under Analytics
        
    Returns:
        None
        
    Examples:
        navigate_to_repository_analytics('612', '456', '458')
    """
    click(project_link_id)  # Click project link
    click(analytics_link_id)  # Click Analytics link
    click(repository_link_id)  # Click Repository link under Analytics

def navigate_to_repository_graphs(project_link_id: str, repo_section_id: str):
    """Navigate from homepage to repository graphs page.
    
    Args:
        project_link_id: The ID of the project link
        repo_section_id: The ID of the repository section link
        
    Returns:
        None
        
    Examples:
        navigate_to_repository_graphs('612', '319')
    """
    click(project_link_id)  # Click project link
    click(repo_section_id)  # Click repository section
    goto("http://172.26.116.102:8080/primer/design/-/graphs/main")  # Navigate to graphs page
```
