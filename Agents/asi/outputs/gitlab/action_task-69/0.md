def set_status_with_keyboard(status_checkbox_id: str, set_button_id: str):
    """Set GitLab status using keyboard navigation when mouse interaction is blocked.
    
    Args:
        status_checkbox_id: The ID of the status checkbox
        set_button_id: The ID of the "Set status" button
        
    Returns:
        None
        
    Examples:
        set_status_with_keyboard('1046', '1095')
    """
    click(status_checkbox_id)  # Focus on status checkbox
    keyboard_press("Tab")  # Navigate to checkbox
    keyboard_press("Space")  # Check the checkbox
    click(set_button_id)  # Click "Set status" button
```
