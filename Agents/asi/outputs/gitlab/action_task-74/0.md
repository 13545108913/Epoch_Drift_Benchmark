```python
def update_project_title(settings_id: str, title_field_id: str, save_button_id: str, new_title: str):
    """Update the project title by accessing settings, filling the title field, and saving changes.
    
    Args:
        settings_id: The ID of the Settings link/button
        title_field_id: The ID of the project title input field
        save_button_id: The ID of the Save changes button
        new_title: The new title to set for the project
        
    Returns:
        None
        
    Examples:
        update_project_title('509', '595', '618', 'GIVE ME SPACE')
        update_project_title('settings_btn', 'title_input', 'save_btn', 'New Project Name')
    """
    click(settings_id)  # Click Settings to access project settings
    fill(title_field_id, new_title)  # Fill the project title field with new title
    click(save_button_id)  # Click Save changes button
```
