## Reusable Functions
```python
def navigate_to_milestones(issues_id: str, milestones_id: str):
    """Navigate from homepage to milestones page.
    
    Args:
        issues_id: The ID of the "Issues" link
        milestones_id: The ID of the "Milestones" link
        
    Returns:
        None
        
    Examples:
        navigate_to_milestones('347', '371')
    """
    click(issues_id)  # Click Issues link
    click(milestones_id)  # Click Milestones link

def create_milestone_form(new_milestone_id: str, title_id: str, start_date_id: str, due_date_id: str, 
                         title: str, start_date: str, due_date: str):
    """Fill out the milestone creation form with title and dates.
    
    Args:
        new_milestone_id: The ID of the "New milestone" button
        title_id: The ID of the title field
        start_date_id: The ID of the start date field
        due_date_id: The ID of the due date field
        title: The milestone title
        start_date: The start date in YYYY-MM-DD format
        due_date: The due date in YYYY-MM-DD format
        
    Returns:
        None
        
    Examples:
        create_milestone_form('559', '551', '626', '633', 'Project Alpha', '2023-03-01', '2023-03-15')
    """
    click(new_milestone_id)  # Click New milestone button
    fill(title_id, title)  # Fill milestone title
    fill(start_date_id, start_date)  # Fill start date
    fill(due_date_id, due_date)  # Fill due date
```
