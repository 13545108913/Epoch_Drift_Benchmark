def navigate_to_profile_settings(preferences_id: str, profile_id: str):
    """Navigate from preferences to profile settings, handling dropdown interference.
    
    Args:
        preferences_id: The ID of the "Go to preferences" link
        profile_id: The ID of the "Profile" link
        
    Returns:
        None
        
    Examples:
        navigate_to_profile_settings('272', '275')
    """
    click(preferences_id)  # Click "Go to preferences" link
    click(profile_id)  # Click "Profile" link to access profile settings

def set_homepage_url(website_field_id: str, url: str, update_button_id: str):
    """Set the homepage URL in the profile settings and save changes.
    
    Args:
        website_field_id: The ID of the Website URL field
        url: The URL to set as homepage
        update_button_id: The ID of the "Update profile settings" button
        
    Returns:
        None
        
    Examples:
        set_homepage_url('878', 'https://example.com', '914')
    """
    fill(website_field_id, url)  # Fill the Website URL field with the provided URL
    click(update_button_id)  # Click the update button to save changes
