```python
def search_project(search_box_id: str | int, project_name: str):
    """Search for a GitLab project using the search box.
    
    Args:
        search_box_id: The ID of the search box element
        project_name: The name of the project to search for
        
    Returns:
        None
        
    Examples:
        search_project('156', 'thoughtbot/administrate')
        search_project('156', 'rails/rails')
    """
    click(search_box_id)  # Click on the search box
    fill(search_box_id, project_name)  # Enter the project name
    keyboard_press("Enter")  # Press Enter to search
```

```python
def navigate_to_contributors(projects_tab_id: str | int, project_link_id: str | int):
    """Navigate to a project's contributor statistics.
    
    Args:
        projects_tab_id: The ID of the "Projects" tab in search results
        project_link_id: The ID of the specific project link to click
        
    Returns:
        None
        
    Examples:
        navigate_to_contributors('313', '415')
        navigate_to_contributors('313', '420')
    """
    click(projects_tab_id)  # Click the Projects tab to filter search results
    click(project_link_id)  # Click on the specific project to view details
```
