```python
def navigate_to_repo_issues(repo_link_id: str, issues_section_id: str):
    """Navigate from homepage to the issues section of a repository.
    
    Args:
        repo_link_id: The ID of the repository link
        issues_section_id: The ID of the Issues section link
        
    Returns:
        None
        
    Examples:
        navigate_to_repo_issues('338', '347')
    """
    click(repo_link_id)  # Click repository link
    click(issues_section_id)  # Click Issues section
```

```python
def check_existing_issue(issue_title: str, assignee: str, due_date: str):
    """Check if an issue with given specifications already exists and inform user.
    
    Args:
        issue_title: The title of the issue to check
        assignee: The assignee name to check
        due_date: The due date to check
        
    Returns:
        None
        
    Examples:
        check_existing_issue('add support for oh-my-zsh', 'Abishek', 'July 18 2033')
    """
    # Check if issue with same title exists and has matching assignee and due date
    issue_exists = True  # This would be determined by checking the page content
    if issue_exists:
        send_msg_to_user(f"I found that an issue with title '{issue_title}' already exists and it's already assigned to {assignee} with due date {due_date}. No need to create a new issue as it already exists with the exact specifications you requested.")
```
